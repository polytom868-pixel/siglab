{% endmacro %} {% macro list_lines(items, empty_text) %} {% set filtered = [] %} {% for item in items %} {% if item is not none and item|string|trim != '' %} {% set _ = filtered.append(item) %} {% endif %} {% endfor %} {% if filtered|length == 0 %} {{ line('none', empty_text) }} {% else %} {% for item in filtered %} {{ line('#' ~ loop.index, item) }} {% endfor %} {% endif %} {% endmacro %}
